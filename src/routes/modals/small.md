```svelte
<script>
import { SmallModal, ModalButton } from "svelte-flow";

  toggleModal(id1, false);
  goto("/dummy-pages/about");
};

  toggleModal(id1, false);
};

// for small modal 2
const btnName2 = "Small Modal 2";
const id2 = "small-modal-2";
const btnColor2 = "red";

const handlebtnS3 = () => {
  toggleModal(id2, false);
};

  toggleModal(id2, false);
};
</script>
```
